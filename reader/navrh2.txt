symbols = {LESS, GREATER, EQUAL, KEY, VALUE, EOF}

s = None  // current symbol
v = None  // symbol value

root()
	next_sym()
	while s != EOF
		if s == ELEMENT_OPENER
			read_element()
		next_sym()


read_element()
	next_sym()
	assert(s == KEY && "tag name expected")
	next_sym()
	if s == KEY
		read_element_attribs()


read_element_attribs()
	while s == KEY
		read_element_attrib()
		next_sym()


read_element_attrib()
	next_sym()
	assert(s == EQUAL && "expected =")
	next_sym()
	assert(s == VALUE && "attribute value expected")


next_sym()
	c = next_char()
	if c == '<'
		s = LESS
	else if c == '>'
		s = GREATER
	else if c == '='
		s = EQUAL
	else
		buf = ''
		if c == '\"'
			s = VALUE
			v = read_until('\"')			
		else
			s = KEY
			v = read_until_space()
